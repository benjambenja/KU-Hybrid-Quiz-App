1. Install npm and nodejs
	sudo apt-get update
	sudo apt-get install nodejs -g
	sudo apt-get install npm -g

2. Install angular
	npm install angular -g

3. Install http-server
	npm install http-server -g

4. Navigate to project directory, run http-server
	http-server
